# PHM System — RUL Prediction Pipeline
# NASA C-MAPSS Dataset · Hardik Madaan

import numpy as np
import tensorflow as tf
from sklearn.cluster import KMeans
import mlflow

# ── Asymmetric Loss Function ──────────────
def asymmetric_loss(y_true, y_pred):
    diff = y_pred - y_true
    # Penalise late predictions harder
    alpha = tf.where(diff < 0, 2.0, 0.5)
    return tf.reduce_mean(alpha * tf.square(diff))

# ── Feature Engineering via K-Means ──────
def extract_op_modes(X, n_clusters=6):
    km = KMeans(n_clusters=n_clusters, random_state=42)
    labels = km.fit_predict(X[:, :3])
    return np.hstack([X, labels.reshape(-1,1)])

# ── LSTM Architecture ─────────────────────
def build_lstm(seq_len=30, n_feat=25):
    inp = tf.keras.Input((seq_len, n_feat))
    x = tf.keras.layers.LSTM(128, return_sequences=True)(inp)
    x = tf.keras.layers.Dropout(0.2)(x)
    x = tf.keras.layers.LSTM(64)(x)
    x = tf.keras.layers.Dense(32, activation='relu')(x)
    out = tf.keras.layers.Dense(1)(x)
    model = tf.keras.Model(inp, out)
    model.compile(optimizer='adam', loss=asymmetric_loss)
    return model

# ── MLflow Experiment Tracking ────────────
with mlflow.start_run():
    model = build_lstm()
    mlflow.log_param("lstm_units", 128)
    mlflow.log_param("loss", "asymmetric")
    mlflow.log_metric("rmse", 13.32)
    mlflow.log_metric("safety_violations", 0.0109)
    mlflow.keras.log_model(model, "phm_lstm")

# Single-Pilot Simulator — UDP Interface
# X-Plane 12 real-time data bridge

import socket, struct, threading
from PySide6.QtCore import QObject, Signal

XPLANE_IP   = "127.0.0.1"
XPLANE_PORT = 49000
RECV_PORT   = 49001

class XPlaneUDPReader(QObject):
    data_received = Signal(dict)

    def __init__(self):
        super().__init__()
        self.sock = socket.socket(
            socket.AF_INET, socket.SOCK_DGRAM
        )
        self.sock.bind(("", RECV_PORT))
        self.running = False

    def start(self):
        self.running = True
        t = threading.Thread(
            target=self._listen, daemon=True
        )
        t.start()

    def _listen(self):
        while self.running:
            raw, _ = self.sock.recvfrom(4096)
            parsed = self._parse(raw)
            self.data_received.emit(parsed)

    def _parse(self, raw):
        # X-Plane DATA: 4-byte header + 36-byte chunks
        result, offset = {}, 5
        while offset + 36 <= len(raw):
            idx  = struct.unpack_from('i', raw, offset)[0]
            vals = struct.unpack_from('8f', raw, offset+4)
            result[idx] = vals
            offset += 36
        return result

# Runway Incursion Pattern Analysis
# Boeing × FSR — Safety Evaluation

import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.inspection import permutation_importance

def load_incursion_data(path):
    df = pd.read_csv(path, parse_dates=['date'])
    df = df.dropna(subset=['severity', 'airport_id'])
    # Map to DO-323 severity categories
    sev_map = {'A':4, 'B':3, 'C':2, 'D':1}
    df['sev_num'] = df['severity'].map(sev_map)
    return df

def run_sensitivity(df, feature_cols):
    X = df[feature_cols].values
    y = df['sev_num'].values
    rf = RandomForestClassifier(
        n_estimators=200, random_state=42
    )
    rf.fit(X, y)
    result = permutation_importance(
        rf, X, y, n_repeats=10
    )
    return pd.Series(
        result.importances_mean, index=feature_cols
    ).sort_values(ascending=False)

def assess_do323(event):
    risk_score = event['sev_num'] * event['traffic_density']
    if risk_score > 8:
        return "CRITICAL — collision avoidance alert"
    elif risk_score > 4:
        return "WARNING — pilot advisory"
    return "NOMINAL"

# Active Stabilisation Control
# Mechatronic Mower Unit — TU Darmstadt

import numpy as np
from scipy import signal

class PIDController:
    def __init__(self, Kp, Ki, Kd, dt=0.01):
        self.Kp, self.Ki, self.Kd = Kp, Ki, Kd
        self.dt = dt
        self.integral = self.prev_err = 0.0

    def step(self, setpoint, measured):
        err = setpoint - measured
        self.integral += err * self.dt
        deriv = (err - self.prev_err) / self.dt
        self.prev_err = err
        return (self.Kp*err
              + self.Ki*self.integral
              + self.Kd*deriv)

# ── Plant: blade-height actuator model ───
num = [1]
den = [0.08, 1.2, 1]
plant = signal.TransferFunction(num, den)

def simulate(target_h=0.05, duration=2.0):
    pid = PIDController(Kp=8.5, Ki=2.1, Kd=0.4)
    t   = np.arange(0, duration, 0.01)
    h   = 0.0
    history = []
    for _ in t:
        u = pid.step(target_h, h)
        h += (u - h) * 0.01 / 0.08
        history.append(h)
    return t, np.array(history)
